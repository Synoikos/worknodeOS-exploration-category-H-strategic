  #!/bin/bash
  AGENT_FILE=$1

    echo "‚ùå Missing: What Was Implemented"
    exit 1
  }

  grep -q "## Critical Decisions" $AGENT_FILE || {
    echo "‚ùå Missing: Critical Decisions"
    exit 1

    echo "‚ùå Missing: Dependencies"
    exit 1

  for f in $FILES; do
    [ -f "src/$f" ] || {
      echo "‚ùå Missing file: src/$f"
      exit 1
    }
  done

  echo "‚úÖ Handoff validated"
